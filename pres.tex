\begin{frame}{Параметра порядка B-фазы $^3$He}
Параметр порядка:
$$
A_{jk} = \Delta(P,T)\ e^{i\phi}\ R_{jk}({\bf n}, \theta).
$$

Дипольная энергия:
$$
F_D = \frac{8 \chi \Omega_B^2}{15\gamma^2}
\left(\cos\theta+\frac14\right)^2.
$$

Минимум достигается при $\theta = \arccos(-1/4) \approx 104.5^\circ$.
\end{frame}


\begin{frame}{Текстура  B-фазы в цилиндрическом сосуде}
 \image{t}{flare_out}{}
 $\alpha = 60^\circ$, $\beta = \cos^{-1}(1/\sqrt{5})\approx 63.4^\circ$
\end{frame}


\begin{frame}{Уравнения Леггетта}
$$
\left\{
\begin{array}{rcl}
\dot{\bf M} &=& \gamma{\bf M} \times {\bf H} +
  \frac{4}{15} \frac{\chi\Omega_B^2}{\gamma} \sin\theta (1+4\cos\theta){\bf n}\\
\dot\theta  &=& \gamma{\bf n}\left(\frac{\bf M}{\chi}-{\bf H}\right)\\
\dot{\bf n} &=& -\frac{\gamma}{2} \left[
  {\bf n} \times \left(\frac{\bf M}{\chi}-{\bf H}\right)+
  \ctg\frac{\theta}{2}\left[
    {\bf n}\left({\bf n}\left(
      \frac{\bf M}{\chi}-{\bf H}
    \right)\right)-\left(
      \frac{\bf M}{\chi}-{\bf H}
    \right)
  \right]
\right]
\end{array}\right.
$$

Частоты малых колебаний при линейном ЯМР:
$$
\omega_1 = \omega_0 + \frac{\Omega_B^2}{2\omega_0}\sin^2\varphi, \qquad
\omega_2 = \Omega_B \cos\varphi,
$$

где $\omega_0 = \gamma H_0$~--- ларморовская частота,
а~$\varphi$~--- угол между~$\bf n$ и~$\bf H$.
\end{frame}

\begin{frame}{Колебания ОПД}
$$
\Omega_1^2 = (\gamma H_0)^2 + \Omega_B^2,
$$
$$
\Omega_2^2 =  \frac{4}{\sqrt{15}} \  \frac{H_R}{H_0}
 \ \frac{\Omega_B^2}{1 + 8/3 (\Omega_B/\gamma H_0)^2},
$$
$$
\Omega_3^2 = \gamma H_0\, \frac{\sqrt{15}\Delta\omega-\gamma H_R}{\sqrt{15}}
 \ \frac{3/8+(\Omega_B/\gamma H_0)^2}{1+(\Omega_B/\gamma H_0)^2}.
$$
 \image{r}{dokl_osc}{}
\end{frame}

\begin{frame}{Численный эксперимент}
\begin{itemize}
\item обнаружение колебаний ОПД, проверка формулы для их частоты;
\item нахождение оптимального способа возбуждения и наблюдения колебаний;
\item изучение влияния спиновой релаксации и пространственной неоднородности на частоту колебаний;
\item изучение влияния объемного $^3$He, находящегося в зазорах между аэрогелем и стенками ячейки.
\end{itemize}
\end{frame}

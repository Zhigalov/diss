\section*{4. Малые колебания ОПД с поглощением}

\hrule\vspace{5mm}

Уравнение Леггетта-Такаги во вращающейся системе координат:

\begin{equation}
\left\{
\begin{array}{ccl}
\dot m_x/\omega_0    &=& - d m_y + bF n_x\\
\dot m_y/\omega_0    &=& h m_z + d m_x + bF n_y\\
\dot m_z/\omega_0    &=& - m_y h + bFn_z\\
\dot\theta/\omega_0  &=& B + kF\\
2\dot n_x/\omega_0 &=& - 2n_y(1+d) - n_y(m_z-1) + n_zm_y - C Bn_x + C(m_x-h)\\
2\dot n_y/\omega_0 &=&   2n_x(1+d) - n_z(m_x-h) + n_x(m_z-1) - C Bn_y + Cm_y\\
2\dot n_z/\omega_0 &=& - n_xm_y + n_y(m_x-h) - C Bn_z + C(m_z-1)\\
B &=& n_x(m_x-h) + n_ym_y + n_z(m_z-1)
\end{array}\right.
\end{equation}

Уравнения для малых отклонений от стационарного решения с поглощением:
\begin{equation}
\left\{
\begin{array}{ccl}
\dot m_x/\omega_0    &=&
    - d \Delta m_y
    + b n_{x0} \Delta F
    + b F_0 \Delta n_x\\
\dot m_y/\omega_0    &=&
      h \Delta m_z
    + d \Delta m_x
    + b n_{y0} \Delta F
    + b F_0 \Delta n_y\\
\dot m_z/\omega_0    &=&
    - h \Delta m_y
    + b n_{z0} \Delta F
    + b F_0 \Delta n_z\\
\dot\theta/\omega_0  &=&
      \Delta B
    + k\Delta F\\
2\dot n_x/\omega_0 &=&
    - 2\Delta n_y(1+d)
    - n_{y0}\Delta m_z
    - (m_{z0}-1)\Delta n_y
    + n_{z0}\Delta m_y
    + m_{y0}\Delta n_z\\
    &-& C_0 B_0\Delta n_x
    - C_0 n_{x0} \Delta B
    - B_0n_{x0} \Delta C
    + (m_{x0}-h)\Delta C
    + C_0\Delta m_x\\
2\dot n_y/\omega_0 &=&
      2\Delta n_x(1+d)
    - n_{z0}\Delta m_x
    - (m_{x0}-h)\Delta n_z
    + n_{x0}\Delta m_z
    + (m_{z0}-1)\Delta n_x\\
    &-& C_0 B_0\Delta n_y
    - C_0 n_{y0} \Delta B
    - B_0n_{y0} \Delta C
    + C_0\Delta m_y
    + m_{y0}\Delta C\\
2\dot n_z/\omega_0 &=&
    - n_{x0}\Delta m_y
    - m_{y0}\Delta n_x
    + n_{y0}\Delta m_x
    + (m_{x0}-h)\Delta n_y
    - C_0 B_0\Delta n_z\\
    &-& C_0 n_{z0} \Delta B
    - B_0n_{z0} \Delta C
    + C_0\Delta m_z
    + (m_{z0}-1)\Delta C\\
\Delta B &=&
      n_{x0}\Delta m_x
    + n_{y0}\Delta m_y
    + n_{z0}\Delta m_z
    + (m_{x0}-h)\Delta n_x
    + m_{y0}\Delta n_y
    + (m_{z0}-1)\Delta n_z\\
\end{array}\right.
\end{equation}

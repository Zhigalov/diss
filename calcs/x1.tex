\section*{1. Уравнения Леггетта-Такаги\\ во вращающейся системе координат}
\hrule\vspace{5mm}

Уравнения Леггетта-Такаги

\begin{equation}
\left\{
\begin{array}{rcl}
\dot{\bf M} &=& \gamma{\bf M} \times {\bf H} +
  \frac{4}{15} \frac{\chi\Omega_B^2}{\gamma} \sin\theta (1+4\cos\theta){\bf n},\\
\dot\theta  &=& \gamma{\bf n}\left(\frac{\bf M}{\chi}-{\bf H}\right) +
  \frac{4}{15\tau}\sin\theta (1+4\cos\theta),\\
\dot{\bf n} &=& -\frac{\gamma}{2} \left[
  {\bf n} \times \left(\frac{\bf M}{\chi}-{\bf H}\right)+
  \ctg\frac{\theta}{2}\left[
    {\bf n}\left({\bf n}\left(
      \frac{\bf M}{\chi}-{\bf H}
    \right)\right)-\left(
      \frac{\bf M}{\chi}-{\bf H}
    \right)
  \right]
\right]
\end{array}\right.
\end{equation}

\hrule\vspace{5mm}

Переход во вращающуюся систему координат.

\begin{equation}
{\bf M}' = R{\bf M}\qquad
{\bf n}' = R{\bf n}\qquad
{\bf H}' = R{\bf H},
\end{equation}

Подставляем в уравнения ${\bf M} = R^{-1}{\bf M'}$,
${\bf n} = R^{-1}{\bf n'}$ и
${\bf H} = R^{-1}{\bf H'}$, умножаем первое и
третье уравнения на $R$ и учитываем, что
$R{\bf a}R{\bf b}={\bf ab}$ и
$R{\bf a} \times R{\bf b} = R ({\bf a}\times {\bf b})$:

\begin{equation}
\left\{
\begin{array}{rcl}
\dot{\bf M'} + R\dot{R}^{-1}{\bf M'} &=& \gamma{\bf M'} \times {\bf H'} +
  \frac{4}{15} \frac{\chi\Omega_B^2}{\gamma} \sin\theta (1+4\cos\theta){\bf n'},\\
\dot\theta  &=& \gamma{\bf n'}\left(\frac{\bf M'}{\chi}-{\bf H'}\right) +
  \frac{4}{15\tau}\sin\theta (1+4\cos\theta),\\
\dot{\bf n'} + R\dot{R}^{-1}{\bf n'} &=& -\frac{\gamma}{2} \left[
  {\bf n'} \times \left(\frac{\bf M'}{\chi}-{\bf H'}\right)+
  \ctg\frac{\theta}{2}\left[
    {\bf n'}\left({\bf n'}\left(
      \frac{\bf M'}{\chi}-{\bf H'}
    \right)\right)-\left(
      \frac{\bf M'}{\chi}-{\bf H'}
    \right)
  \right]
\right]
\end{array}\right.
\end{equation}

\hrule\vspace{5mm}

Пусть ${\bf H} = \left( H_R\cos(\omega t), -H_R\sin(\omega t), H_0\right)$:
постоянное поле $H_0$ вдоль оси $z$ и перпендикулярное ему поле
$H_R$ радиочастотной накачки на частоте $\omega$. Перейдем в систему
координат, в котором $\bf H'$ неподвижно. При этом

\begin{equation}
R = \left(
\begin{array}{ccc}
\cos(\omega t) & -\sin(\omega t) & 0\\
\sin(\omega t) & \cos(\omega t) & 0\\
0&0&1
\end{array}\right), \qquad
R\dot{R}^{-1} = \omega \left(
\begin{array}{ccc}
0  & 1 & 0\\
-1 & 0 & 0\\
0  & 0 & 0
\end{array}\right).
\end{equation}

Введем обозначения

$$
{\bf m'} = \frac{\bf M'}{\chi H_0},\quad
\gamma H_0 = \omega_0,\qquad
F = \frac{4}{15} \sin\theta (1+4\cos\theta),\qquad
C = \ctg\frac{\theta}{2},\qquad
b = \frac{\Omega_B^2}{\omega_0^2}
$$

и опустим штрихи:

\begin{equation}
\left\{
\begin{array}{rcl}
\dot{\bf m} &=& - R\dot{R}^{-1}{\bf m} +
  \omega_0 {\bf m} \times \frac{\bf H}{H_0} + \omega_0 b F{\bf n},\\
\dot\theta  &=& \omega_0{\bf n}\left({\bf m}-\frac{\bf H}{H_0}\right) +
  \frac{F}{\tau},\\
\dot{\bf n} &=& - R\dot{R}^{-1}{\bf n} - \frac{\omega_0}{2} \left[
  {\bf n} \times \left({\bf m}-\frac{\bf H}{H_0}\right)+
  C\left[
    {\bf n}\left({\bf n}\left(
      {\bf m}-\frac{\bf H}{H_0}
    \right)\right)-\left(
      {\bf m}-\frac{\bf H}{H_0}
    \right)
  \right]
\right]
\end{array}\right.
\end{equation}

\hrule\vspace{5mm}\eject

Распишем уравнения в координатах:

\begin{equation}
\left\{
\begin{array}{ccl}
\dot m_x    &=& - m_y\omega + m_y\omega_0 + \omega_0 b Fn_x\\
\dot m_y    &=&   m_x\omega + m_z\gamma H_R - M_x\omega_0 + \omega_0bFn_y\\
\dot m_z    &=& - m_y\gamma H_R + \omega_0bFn_z\\
\dot\theta  &=& B \omega_0 + F/\tau,\\	
\dot n_x &=& - n_y\omega -\frac{\omega_0}{2} \left[
	n_y(m_z-1) - n_zm_y + C\{ Bn_x -(m_x-\gamma H_R/\omega_0)\}
\right]\\
\dot n_y &=& n_x\omega -\frac{\omega_0}{2} \left[
    n_z(m_x-\gamma H_R/\omega_0) - n_x(m_z-1) + C\{ Bn_y - m_y \}
\right]\\
\dot n_z &=& -\frac{\omega_0}{2} \left[
    n_xm_y - n_y(m_x-\gamma H_R/\omega_0) + C\{ Bn_z - (m_z-1) \}
\right]
\end{array}\right.
\end{equation}

где $B = {\bf n}\left({\bf m}-\frac{\bf H}{H_0}\right) =
n_x(m_x-\gamma H_R/\omega_0) + n_ym_y + n_z(m_z-1)$.

Введем безразмерный сдвиг частоты $d$, радиочастотное поле $h$ и
коэффициент поглощения $k$:

$$
d = \frac{\omega-\omega_0}{\omega_0},\quad
h =\frac{\gamma H_R}{\omega_0},\quad
k = \frac1{\omega_0\tau}
$$

\section*{3. Стационарное решение с учетом поглощения}

\hrule\vspace{5mm}

Стационарные уравнения Леггетта-Такаги во вращающейся системе координат:

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=& - d m_y + bF n_x\\
0 &=& h m_z + d m_x + bF n_y\\
0 &=& - m_y h + bFn_z\\
0 &=& B + kF\\
0 &=& - 2n_y(1+d) - n_y(m_z-1) + n_zm_y - C Bn_x + C(m_x-h)\\
0 &=&   2n_x(1+d) - n_z(m_x-h) + n_x(m_z-1) - C Bn_y + Cm_y\\
0 &=& - n_xm_y + n_y(m_x-h) - C Bn_z + C(m_z-1)\\
B &=& n_x(m_x-h) + n_ym_y + n_z(m_z-1)
\end{array}\right.
\end{equation}

Для отклонений от случая без поглощения:

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=& - d \Delta m_y + b F_0 \Delta n_x\\
0 &=&   h \Delta m_z + d \Delta m_x + b F_0 \Delta n_y + b n_{y0} \Delta F \\
0 &=& - h \Delta m_y + b F_0 \Delta n_z\\
0 &=&   (m_{x0}-h)\Delta n_x + n_{y0} \Delta m_y + (m_{z0}-1)\Delta n_z + k F_0\\
0 &=& - 2(1+d)\Delta n_y - n_{y0} \Delta m_z - (m_{z0}-1)\Delta n_y + C_0 \Delta m_x + (m_{x0}-h)\Delta C\\
0 &=&   2(1+d)\Delta n_x - (m_{x0}-h)\Delta n_z + (m_{z0}-1)\Delta n_x - C_0 n_{y0}\Delta B + C_0 \Delta m_y\\
0 &=&   (m_{x0}-h)\Delta n_y + n_{y0} \Delta m_x + C_0\Delta m_z + (m_{z0}-1)\Delta C\\
\Delta B &=&  (m_{x0}-h)\Delta n_x + n_{y0} \Delta m_y + (m_{z0}-1)\Delta n_z\\
\end{array}\right.
\end{equation}

--- ответвление: кажется, что нельзя выкидывать члены младших порядков, проверяю 

Из первого и третьего уравнений:

\begin{equation}
\Delta n_x=\frac{d}{bF_0} \Delta m_y,\qquad
\Delta n_z=\frac{h}{bF_0} \Delta m_y,\qquad
\end{equation}

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=&
   h \Delta m_z
 + d \Delta m_x
 + b F_0 \Delta n_y
 + b s_1 \Delta F
\\
0 &=&
 \left[
   (m_{x0}-h)\frac{d}{bF_0} + (m_{z0}-1) \frac{h}{bF_0} + s_1
 \right] \Delta m_y
 + k F_0
\\
0 &=&
 - 2(1+d)\Delta n_y
 - s_1\Delta m_z
 - (m_{z0}-1)\Delta n_y
 + C_0 \Delta m_x
 + (m_{x0}-h)\Delta C
\\
0 &=&
 \left[
   2(1+d)\frac{d}{bF_0} - (m_{x0}-h)\frac{h}{bF_0} + (m_{z0}-1)\frac{d}{bF_0}
  - C_0 s_1 (m_{x0}-h)\frac{d}{bF_0} - C_0 s_1 (m_{z0}-1)\frac{h}{bF_0}
 \right] \Delta m_y
\\
0 &=&
   (m_{x0}-h)\Delta n_y
 + s_1 \Delta m_x
 + C_0\Delta m_z
 + (m_{z0}-1)\Delta C
\\
\end{array}\right.
\end{equation}

В предпоследнем уравнении перед $\Delta m_y$ стоит 0 (?)

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=&
   h \Delta m_z

\\
\Delta m_y &=& k \Theta_0

\end{equation}

считаю что $\Delta n_y = 0$ (из того, что $|{\bf n}|=1$).

--- конец ответвления

Оставляем в каждом уравнении только члены старшего порядка по $d$ и $h$:

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=& - d \Delta m_y + b F_0 \Delta n_x\\
0 &=&   b n_{y0} \Delta F \\
0 &=& - h \Delta m_y + b F_0 \Delta n_z\\
0 &=&   m_{x0}\Delta n_x + n_{y0} \Delta m_y + (m_{z0}-1)\Delta n_z + k F_0\\
0 &=& - n_{y0}\Delta m_z - (m_{z0}+1)\Delta n_y + C_0 \Delta m_x + m_{x0}\Delta C\\
0 &=& - m_{x0}\Delta n_z + (m_{z0}+1)\Delta n_x - C_0 n_{y0}\Delta B + C_0 \Delta m_y\\
0 &=&   m_{x0}\Delta n_y + n_{y0} \Delta m_x + C_0\Delta m_z + (m_{z0}-1)\Delta C\\
\Delta B &=&  m_{x0}\Delta n_x + n_{y0} \Delta m_y + (m_{z0}-1)\Delta n_z\\
\end{array}\right.
\end{equation}

Из первых трех уравнений:

\begin{equation}
\Delta n_x=\frac{d \Delta m_y}{bF_0},\qquad
\Delta n_z=\frac{h \Delta m_y}{bF_0},\qquad
\Delta F = 0, \quad \Delta \Theta=0,\quad \Delta C = 0.
\end{equation}

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=&  \Delta m_y\left[ m_{x0}\frac{d}{bF_0} + n_{y0} -\frac54 \frac{h}{bF_0}\right] + k F_0\\
0 &=& - n_{y0}\Delta m_z - \frac34\Delta n_y + C_0 \Delta m_x\\
0 &=& \frac{\Delta m_y}{bF_0}\left[ - m_{x0}h + \frac34 d - C_0 n_{y0} m_{x0}d - C_0 n_{y0}\frac34 h \right]\\
0 &=&   m_{x0}\Delta n_y + n_{y0} \Delta m_x + C_0\Delta m_z
\end{array}\right.
\end{equation}

Подставляем равновесные значения без поглощения:

\begin{equation}
\left\{
\begin{array}{ccl}
0 &=&  \Delta m_y\left[ s\frac{\sqrt{15}}4 \frac{d}{-bs_2\Theta} + s_1 - \frac54 \frac{h}{-bs_2\Theta}\right] - k s_2\Theta\\
0 &=& - s_1 \Delta m_z - \frac34\Delta n_y + s_2\sqrt{\frac35} \Delta m_x\\
0 &=& \frac{\Delta m_y}{-bs_2\Theta}\left[ - s\frac{\sqrt{15}}4 h + \frac34 d - s_2\sqrt{\frac35} s_1 s\frac{\sqrt{15}}4 d - s_2\sqrt{\frac35} s_1 \frac34 h \right]\\
0 &=&   s\frac{\sqrt{15}}4\Delta n_y + s_1 \Delta m_x + s_2\sqrt{\frac35} \Delta m_z
\end{array}\right.
\end{equation}

Выражение в квадратных скобках в третьем уравнении
обращается 0, а из первого уравнения получается

\begin{equation}
\Delta m_y =  \frac{kb \Theta_0^2}{h}
\end{equation}

Отсюда
\begin{equation}
\Delta n_x=-\frac{d}{h}k\Theta_0,\qquad
\Delta n_z=-k\Theta_0.
\end{equation}

Из условия $n_x^2 + n_y^2 + n_z^2 = 1$

\begin{equation}
\Delta n_x^2 + \Delta n_z^2 + 2\Delta n_y = 0, \qquad
\Delta n_y = -k^2\Theta_0^2 \frac{1+(d/h)^2}{2}.
\end{equation}

Из оставшиеся уравнений видно, что и
$\Delta m_x, \Delta m_z, \Delta \Theta \sim k^2\Theta_0^2$
